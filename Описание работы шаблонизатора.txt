${Имя(атрибуты,атрибут:"автоинкремент_шаблон";константы=)}
$[
код с вставкой $(атрибутов)
]$


"автоинкремент_шаблон"
- содержит либо i в качестве локального id
- либо глобальную переменную
- в любом случае, определяется eval-ом
- ; заменяется на перенос строки в python


ключевое слово "$$" - определяет, где находится текущий курсор в файле (куда вставляются все шаблонизированные компоненты)


Инициализация
- для создание своего шаблона, обязательно необходимо добавить каждый из элементов, что планируется использовать в шаблоне (ТК они создают в соответствующем .inx, .cir и прочих файлах общие константы для всех компонентов (например отрисовку))



Пример
${Or2(x,y,id:"'global ui;return '")}
$[
[Comp]
Name=Or2
Px=$(x),$(y)

[Attr]
ON=15,4,PART
V=U2
Style=8192

[Attr]
ON=15,18,TIMING MODEL
V=D0_GATE

[Attr]
ON=15,32,I/O MODEL
V=IO_STD

[Attr]
ON=15,46,MNTYMXDLY
V=0

[Attr]
ON=15,60,IO_LEVEL
V=0

[Attr]
ON=15,74,POWER NODE
V=$G_DPWR

[Attr]
ON=15,88,GROUND NODE
V=$G_DGND

[Attr]
ON=0,0,PACKAGE
V
]$
